PREFIX dblp: <https://dblp.org/rdf/schema#>

SELECT ?year (COUNT(?pub) AS ?count) WHERE {
  # We filter by the specific "streams" (conference series) that define AI
  VALUES ?venue {
    <https://dblp.org/streams/conf/nips>   # NeurIPS (Neural Information Processing Systems)
    <https://dblp.org/streams/conf/icml>   # ICML (International Conf. on Machine Learning)
    <https://dblp.org/streams/conf/cvpr>   # CVPR (Computer Vision)
    <https://dblp.org/streams/conf/icc>    # ICCV (Computer Vision)
    <https://dblp.org/streams/conf/aaai>   # AAAI (Assoc. for the Advancement of AI)
    <https://dblp.org/streams/conf/ijcai>  # IJCAI (Intl. Joint Conf. on AI)
    <https://dblp.org/streams/conf/acl>    # ACL (Computational Linguistics / NLP)
  }

  ?pub dblp:publishedInStream ?venue .
  ?pub dblp:yearOfPublication ?year .
  
}

PREFIX dblp: <https://dblp.org/rdf/schema#>
PREFIX cito: <http://purl.org/spar/cito/>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>

SELECT ?subtopic (COUNT(?paper) AS ?paperCount) (SUM(?cites) AS ?totalCitations)
WHERE {
  {
    SELECT ?paper (COUNT(?citation) AS ?cites) WHERE {
      ?citation cito:hasCitedEntity ?omid .
      ?paper dblp:omid ?omid .
    } GROUP BY ?paper
  }
  
  ?paper dblp:title ?title .
  
  BIND(
    IF(REGEX(?title, "Transformer|LLM|Large Language Model|GPT|Attention Mechanism", "i"), "Generative AI & LLMs",
    IF(REGEX(?title, "Diffusion Model|Stable Diffusion|GAN|Generative Adversarial", "i"), "Image & Media Synthesis",
    IF(REGEX(?title, "Reinforcement Learning|RLHF|Policy Gradient", "i"), "Reinforcement Learning",
    IF(REGEX(?title, "Computer Vision|Object Detection|Segmentation|Image Recognition", "i"), "Computer Vision",
    IF(REGEX(?title, "Natural Language|NLP|Sentiment Analysis|Text Mining", "i"), "Classical NLP",
    IF(REGEX(?title, "Explainable AI|XAI|Interpretability|Fairness|Ethics|Bias", "i"), "AI Ethics & Interpretability",
    IF(REGEX(?title, "Federated Learning|Privacy-Preserving|Differential Privacy", "i"), "Privacy & Distributed AI",
    IF(REGEX(?title, "Graph Neural Network|GNN|Relational Learning", "i"), "Graph Learning",
    IF(REGEX(?title, "Edge AI|TinyML|Hardware Acceleration|Neuromorphic", "i"), "AI Hardware & Edge",
    "General AI/Other"))))))))) AS ?subtopic
  )
  
  FILTER(?subtopic != "General AI/Other")
}
GROUP BY ?subtopic
ORDER BY DESC(?totalCitations)

SELECT ?subtopic 
       (MIN(xsd:integer(STR(?year))) AS ?birthYear) 
       (AVG(xsd:decimal(STR(?year))) AS ?avgYear)
       (COUNT(?paper) AS ?totalVolume)
WHERE {
  ?paper a dblp:Publication ;
         dblp:title ?title ;
         dblp:yearOfPublication ?year .
  
  # Optimization: Filter was changed by hand multiple times to get results
  FILTER(xsd:integer(STR(?year)) > 2010 && xsd:integer(STR(?year)) <= 2020)

  BIND(
    IF(REGEX(?title, "Transformer|Large Language Model|LLM|GPT-[34]|Attention Mechanism", "i"), "LLMs & Attention",
    IF(REGEX(?title, "Diffusion Model|Stable Diffusion|DALL-E|Generative Adversarial", "i"), "Generative Media",

Part 1 - Affiliation

PREFIX dblp: <https://dblp.org/rdf/schema#>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>

SELECT ?subtopic ?hasAffiliation (COUNT(?paper) AS ?count)
WHERE {
  ?paper a dblp:Publication ;
         dblp:title ?title ;
         dblp:yearOfPublication ?year .

  BIND(
    IF(REGEX(?title, "Large Language Model|LLM|Transformer", "i"), "Generative AI",
    IF(REGEX(?title, "Reinforcement Learning", "i"), "RL",
    IF(REGEX(?title, "Computer Vision", "i"), "Vision",
    "Other"))) AS ?subtopic
  )
  
  FILTER(?subtopic != "Other")

  OPTIONAL { 
    ?paper dblp:author ?author . 
    ?author dblp:note ?note . 
    FILTER(CONTAINS(LCASE(STR(?note)), "affiliation")) 
  }
  
  BIND(IF(BOUND(?note), "With Affiliation", "Missing Affiliation") AS ?hasAffiliation)
}
GROUP BY ?subtopic ?hasAffiliation
ORDER BY ?subtopic

Part 2 - Citations

PREFIX dblp: <https://dblp.org/rdf/schema#>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>

SELECT ?subtopic ?citationStatus (COUNT(?paper) AS ?count)
WHERE {
  ?paper a dblp:Publication ;
         dblp:title ?title ;
         dblp:yearOfPublication ?year .

  BIND(
    IF(REGEX(?title, "Large Language Model|LLM|Transformer", "i"), "Generative AI",
    IF(REGEX(?title, "Reinforcement Learning", "i"), "RL",
    IF(REGEX(?title, "Computer Vision", "i"), "Vision",
    "Other"))) AS ?subtopic
  )
  FILTER(?subtopic != "Other")

  OPTIONAL { ?paper dblp:omid ?omid . }
  
  BIND(IF(BOUND(?omid), "Citations Tracked", "No Citation Data") AS ?citationStatus)
}
GROUP BY ?subtopic ?citationStatus
ORDER BY ?subtopic
